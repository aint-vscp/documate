<objective>
Create and verify release tag v1.0.0 with deterministic safety checks.

Output: validated release tag and reproducible tag evidence.
</objective>

<context>

@.git
</context>

<tasks>

<task type="auto" tdd="false">
  <name>Task 1: Confirm clean working tree before tagging</name>
  <read_first>
    - git status
  </read_first>
  <files>
    - .git
  </files>
  <action>
    Ensure no staged or unstaged changes exist before creating release tag.
  </action>
  <verify>
    <automated>powershell -NoProfile -Command "$ErrorActionPreference='Stop'; $s = git status --short; if ($s) { throw 'working-tree-not-clean' }; Write-Host 'clean-tree-ok'"</automated>
  </verify>
  <acceptance_criteria>
    Working tree is clean and safe for release tagging.
  </acceptance_criteria>
</task>

<task type="auto" tdd="false">
  <name>Task 2: Confirm tag does not already exist</name>
  <read_first>
    - git tag list
  </read_first>
  <files>
    - .git
  </files>
  <action>
    Verify v1.0.0 is not already present to prevent accidental retagging.
  </action>
  <verify>
    <automated>powershell -NoProfile -Command "$ErrorActionPreference='Stop'; $t = git tag --list v1.0.0; if ($t) { throw 'tag-already-exists' }; Write-Host 'tag-availability-ok'"</automated>
  </verify>
  <acceptance_criteria>
    Tag name v1.0.0 is available.
  </acceptance_criteria>
</task>

<task type="auto" tdd="false">
  <name>Task 3: Create annotated tag and verify resolution</name>
  <read_first>
    - git rev-parse HEAD
  </read_first>
  <files>
    - .git
  </files>
  <action>
    Create annotated tag v1.0.0 and verify it resolves to the intended release commit.
  </action>
  <verify>
    <automated>powershell -NoProfile -Command "$ErrorActionPreference='Stop'; git tag -a v1.0.0 -m 'DocuMate v1.0.0'; git show v1.0.0 --no-patch --oneline | Out-Null; Write-Host 'release-tag-ok'"</automated>
  </verify>
  <acceptance_criteria>
    Annotated tag v1.0.0 exists and is verifiable.
  </acceptance_criteria>
</task>

</tasks>

<verification>
- powershell -NoProfile -Command "$ErrorActionPreference='Stop'; git show v1.0.0 --no-patch --oneline | Out-Null; Write-Host 'phase-07-03-tag-verify-ok'"
</verification>
